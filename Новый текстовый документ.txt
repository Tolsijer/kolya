общее кол во бандеролей полученных путиным

SELECT SUM(dbo.Название_типа)
FROM dbo.Типы_отправлений
WHERE dbo.Получатель.Фамилия = 'Путин'



SELECT        dbo.Получатель.Фамилия, dbo.Типы_отправлений.Название_типа
FROM            dbo.Отправление INNER JOIN
                         dbo.Получатель ON dbo.Отправление.id_получателя = dbo.Получатель.id_получателя INNER JOIN
                         dbo.Типы_отправлений ON dbo.Отправление.id_типа = dbo.Типы_отправлений.id_типа
WHERE        (dbo.Получатель.Фамилия = N'Путин') AND (dbo.Типы_отправлений.Название_типа = N'Бандероль')

SELECT    COUNT(dbo.Получатель.Фамилия, dbo.Типы_отправлений.Название_типа)
FROM            dbo.Отправление INNER JOIN
                         dbo.Получатель ON dbo.Отправление.id_получателя = dbo.Получатель.id_получателя INNER JOIN
                         dbo.Типы_отправлений ON dbo.Отправление.id_типа = dbo.Типы_отправлений.id_типа
WHERE        (dbo.Получатель.Фамилия = N'Путин') AND (dbo.Типы_отправлений.Название_типа = N'Бандероль')
